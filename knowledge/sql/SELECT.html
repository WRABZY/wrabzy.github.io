SELECT

Простая выборка

SELECT column1, column2, ... FROM table;
SELECT * FROM table;

Выборка с сортировкой

SELECT col1, col2, ... FROM table ORDER BY col1, 2, col3;
SELECT col1, col2, ... FROM table ORDER BY col1 DESC, col3;

Выборка с фильтром

SELECT col1, col2, ... FROM table WHERE col1 = 5;

Операции фильтрации

LIKE и метасимволы

Вычисляемые поля

Конкатенация

SELECT col1 || ', ' || col2 FROM table; SELECT col1 + ', ' + col2 FROM table; SELECT CONCAT(col1, ', ', col2) FROM table;

Тримминг пробелов

SELECT RTRIM(col1) FROM table; SELECT LTRIM(col1) FROM table; SELECT TRIM(col1) FROM table;

Псевдонимы

SELECT col1 AS firstCol FROM table;

Математика

SELECT col1 + col2 FROM table; SELECT col1 - col2 FROM table; SELECT col1 * col2 FROM table; SELECT col1 / col2 FROM table;

Группировка

SELECT column1, column2, ... FROM table GROUP BY column1;
SELECT column1, column2, ... FROM table GROUP BY column1 HAVING COUNT(*) >= 2;
