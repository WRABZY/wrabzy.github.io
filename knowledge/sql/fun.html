Функции

Для строк

Тримминг пробелов

SELECT RTRIM(col1) FROM table; SELECT LTRIM(col1) FROM table; SELECT TRIM(col1) FROM table;

Регистр

SELECT UPPER(col1) FROM table; SELECT LOWER(col1) FROM table; SELECT UCASE(col1) FROM table; SELECT LCASE(col1) FROM table;

Длина

SELECT LENGTH(col1) FROM table; SELECT LEN(col1) FROM table; SELECT DATALENGTH(col1) FROM table;

SOUNDEX

Дата / время

Часть даты

SELECT col1 FROM table WHERE DATE_PART('year', col2) = 2021; SELECT col1 FROM table WHERE DATEPART(yy, col2) = 2021; SELECT col1 FROM table WHERE DATEPART('yyyy', col2) = 2021; SELECT col1 FROM table WHERE YEAR(col2) = 2021; SELECT col1 FROM table WHERE to_number(to_char(col2, 'YY')) = 2021; SELECT col1 FROM table WHERE col2 BETWEEN to_date('01-JAN-2021') AND to_date('31-DEC-2021');

Числа

SELECT ABS(col1) FROM table; SELECT COS(col1) FROM table; SELECT EXP(col1) FROM table; SELECT PI(col1) FROM table; SELECT SIN(col1) FROM table; SELECT SQRT(col1) FROM table; SELECT TAN(col1) FROM table;

Статистика

SELECT AVG(col1) FROM table; SELECT COUNT(*) FROM table; SELECT COUNT(col1) FROM table; SELECT MAX(col1) FROM table; SELECT MIN(col1) FROM table; SELECT SUM(col1) FROM table;

ALL и DISTINCT

ALL(подсчет всего) не требуется указывать, подразумевается по умолчанию. Пример использования DISTINCT(подсчет только уникальных): SELECT COUNT(DISTINCT col1) FROM table;
